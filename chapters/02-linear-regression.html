# Linear Regression & Model Selection

```{r}
#| include: false
library(dplyr)
library(tidyr)
library(ggplot2)
library(plotly)
library(scales)

# ── Metric helper functions ───────────────────────────────────────────────────
R2_fn   <- function(y, yhat) { rss <- sum((y - yhat)^2); tss <- sum((y - mean(y))^2); 1 - rss/tss }
MSE_fn  <- function(y, yhat) mean((y - yhat)^2)
RMSE_fn <- function(y, yhat) sqrt(MSE_fn(y, yhat))
MAE_fn  <- function(y, yhat) mean(abs(y - yhat))

# ── Q1: Insurance Charges — nested models ────────────────────────────────────
insurance <- read.csv("Files/Insurance.csv")
insurance <- insurance[!is.na(insurance$charges), ]

set.seed(42)
ins_n         <- nrow(insurance)
ins_train_idx <- sample(1:ins_n, size = floor(0.8 * ins_n), replace = FALSE)
ins_train     <- insurance[ins_train_idx, ]
ins_valid     <- insurance[-ins_train_idx, ]

ins_m1 <- lm(charges ~ age,                                    data = ins_train)
ins_m2 <- lm(charges ~ age + bmi,                              data = ins_train)
ins_m3 <- lm(charges ~ age + bmi + smoker,                     data = ins_train)
ins_m4 <- lm(charges ~ age + bmi + smoker + children,          data = ins_train)
ins_m5 <- lm(charges ~ age + sex + bmi + children + smoker + region, data = ins_train)
ins_models <- list(ins_m1, ins_m2, ins_m3, ins_m4, ins_m5)

ins_labels <- c("age", "age+bmi", "age+bmi+smoker",
                "age+bmi+smoker+children", "age+sex+bmi+children+smoker+region")

ins_train_r2     <- sapply(ins_models, function(m) R2_fn(ins_train$charges, predict(m, ins_train)))
ins_train_r2_adj <- sapply(ins_models, function(m) summary(m)$adj.r.squared)

ins_val_metrics <- data.frame(
  Model           = ins_labels,
  Train_R2        = round(ins_train_r2, 4),
  Train_R2_adj    = round(ins_train_r2_adj, 4),
  Validation_MSE  = NA_real_,
  Validation_RMSE = NA_real_,
  Validation_MAE  = NA_real_
)
for (i in seq_along(ins_models)) {
  yhat <- predict(ins_models[[i]], newdata = ins_valid)
  ins_val_metrics$Validation_MSE[i]  <- round(MSE_fn(ins_valid$charges,  yhat), 0)
  ins_val_metrics$Validation_RMSE[i] <- round(RMSE_fn(ins_valid$charges, yhat), 2)
  ins_val_metrics$Validation_MAE[i]  <- round(MAE_fn(ins_valid$charges,  yhat), 2)
}
best_ins_idx <- which.min(ins_val_metrics$Validation_MSE)

# ── Q2: Used Car Price ────────────────────────────────────────────────────────
val_cars <- read.csv("https://raw.githubusercontent.com/maxwellhahn/maxwellhahn/refs/heads/main/Validation_set.csv")
val_cars <- val_cars[!is.na(val_cars$Miles) & !is.na(val_cars$Price), ]
val_cars$ModelTacoma     <- ifelse(val_cars$Model == "Tacoma", 1, 0)
val_cars$ModelVolt       <- ifelse(val_cars$Model == "Volt",   1, 0)
val_cars$Predicted_Price <- 18.69496 - 0.07807 * val_cars$Miles +
                            9.45872 * val_cars$ModelTacoma -
                            11.02341 * val_cars$ModelVolt
cars_rmse <- round(sqrt(mean((val_cars$Price - val_cars$Predicted_Price)^2)), 4)

# ── Q3-Q5: NOVA Housing EDA & Regression ─────────────────────────────────────
nova4 <- read.csv("Files/NOVA.csv")
nova4_clean <- nova4[nova4$Price <= 3000000 &
                     !(nova4$CountySchool %in% c("LOUDOUN COUNTY PUBLIC SCHOOLS",
                                                  "PRINCE WILLIAM COUNTY PUBLIC SCHOOLS")), ]
nova4_clean$logPrice  <- log(nova4_clean$Price)
nova4_clean$Single    <- ifelse(nova4_clean$ListingType == "Single Family", 1, 0)
nova4_clean$Arlington <- ifelse(nova4_clean$CountySchool == "ARLINGTON COUNTY PUBLIC SCHOOLS", 1, 0)
nova4_outliers <- nova4[nova4$Price > 3000000, ]

model_bath  <- lm(logPrice ~ Bathrooms,             data = nova4_clean)
model_multi <- lm(logPrice ~ Bathrooms + Arlington, data = nova4_clean)

bath_cor   <- round(cor(nova4_clean$logPrice, nova4_clean$Bathrooms), 4)
bath_r2    <- round(summary(model_bath)$r.squared, 4)
bath_sigma <- round(summary(model_bath)$sigma, 5)
bath_slope <- round(coef(model_bath)["Bathrooms"], 6)
bath_int   <- round(coef(model_bath)["(Intercept)"], 6)

multi_r2     <- round(summary(model_multi)$r.squared, 4)
multi_r2_adj <- round(summary(model_multi)$adj.r.squared, 4)
multi_bath_b <- round(coef(model_multi)["Bathrooms"], 6)
multi_arl_b  <- round(coef(model_multi)["Arlington"], 6)
multi_int    <- round(coef(model_multi)["(Intercept)"], 6)
bath_ci90    <- confint(model_multi, parm = "Bathrooms", level = 0.90)

max_bath_idx  <- which.max(nova4_clean$Bathrooms)
max_bath_obs  <- nova4_clean[max_bath_idx, ]
max_bath_pred <- predict(model_bath, newdata = max_bath_obs)
```

| M5 | all vars | `r ins_val_metrics$Validation_MSE[5]` | `r ins_val_metrics$Validation_RMSE[5]` | `r ins_val_metrics$Validation_MAE[5]` |

`smoker` is the single most important predictor — adding it in M3 cuts validation MSE dramatically. Beyond M4, adding sex and region yields no meaningful improvement on unseen data.

## Key Findings — Multiple Regression

**Model:** logPrice = `r multi_int` + `r multi_bath_b`(Bathrooms) + `r multi_arl_b`(Arlington)

|---|---|
